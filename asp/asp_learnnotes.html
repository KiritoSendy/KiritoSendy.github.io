文件下载, 模板下载功能

/// <summary>
/// 浏览器下载文件
/// </summary>
/// <param name="modifyName">自定义文件名, 附带后缀</param>
public void DownloadFile(string modifyName)
{
    // 定义变量: 文件名
    string fileName = string.Empty;

    // 获取文件在服务器的地址
    string url = Server.MapPath("../searchZclbList.xls");

    // 判断地址是否为空
    if (string.IsNullOrEmpty(url))
    {
        // 提示 "该文件暂不提供下载"
        Page.ClientScript.RegisterStartupScript(Page.GetType(), "message", "<script defer>alert('该文件暂不提供下载!');</script>");
        return;
    }

    // 判断获取的是否为地址  
    if (url.IndexOf("\\") > -1)
    {
        // url 为地址时, 则截取 url 中的文件名作为文件名
        fileName = url.Substring(url.LastIndexOf("\\") + 1);
    }
    else
    {
        // url 为文件名时, 直接获取文件名  
        fileName = url;
    }

    // 自定义文件名, 附带后缀
    if (!string.IsNullOrEmpty(modifyName))
    {
        fileName = modifyName;
    }

    // 以字符流的方式下载文件  
    FileStream fileStream = new FileStream(@url, FileMode.Open);
    byte[] bytes = new byte[(int)fileStream.Length];
    fileStream.Read(bytes, 0, bytes.Length);
    fileStream.Close();
    Response.ContentType = "application/octet-stream";

    // 通知浏览器下载 
    Response.AddHeader("Content-Disposition", "attachment; filename=" + fileName);
    Response.BinaryWrite(bytes);
    Response.Flush();
    Response.End();
}

最基本的图片查看器功能

// 前台
<asp:Image ID="Image" runat="server" ImageUrl="~/imagelook.jpg"/>

// 后台
protected void Page_Load(object sender, System.EventArgs e)
{
    try
    {
        // 读取图片
        int ImgID = Convert.ToInt32(Request.QueryString["imageid"] as string);
        OracleConnection Con = new googoSoftEAM.SQLServerDAL.SqlHelper().Connection;
        String SqlCmd = "select * from zc_image where imageid = :imageid and saasdm='" + Session["saasdm"].ToString() + "'";
        OracleCommand CmdObj = new OracleCommand(SqlCmd, Con);
        CmdObj.Parameters.Add(":ImageID", OracleType.UInt32).Value = ImgID;
        Con.Open();
        OracleDataReader SqlReader = CmdObj.ExecuteReader();
        SqlReader.Read();

        // 图片类型
        string imageType = (string)SqlReader["imagecontenttype"];

        // 原版图片内容 [字节流]
        MemoryStream bufferByte = new MemoryStream((byte[])SqlReader["imageData"]);
        System.Drawing.Image imageOrigin = System.Drawing.Image.FromStream(bufferByte, true);

        // 缩小文件 (计算缩小倍率) 缩小至 高度 targetHeight 左右
        float targetHeight = 600.0f;
        float shrink = targetHeight / imageOrigin.Height;
        float w = shrink, h = shrink;

        // 目标图形
        System.Drawing.Bitmap imageSmall = new System.Drawing.Bitmap((int)(imageOrigin.Width * shrink), (int)(imageOrigin.Height * shrink));

        // 创建一个图形对象
        Graphics graphics = Graphics.FromImage(imageSmall);
        graphics.DrawImage(imageOrigin, new Rectangle(0, 0, imageSmall.Width, imageSmall.Height), new Rectangle(0, 0, imageOrigin.Width, imageOrigin.Height), GraphicsUnit.Pixel);

        string ret_type = string.Empty;
        GetImageType(imageType, out ret_type);
        string imagePath = Server.MapPath("../imagelook." + ret_type);
        if (File.Exists(imagePath))
        {
            File.Delete(imagePath);
        }

        imageSmall.Save(imagePath);
        Image.ImageUrl = "~/imagelook." + ret_type + "?" + new Random().Next(100).ToString();
        Con.Close();
    }
    catch (Exception ex)
    {
        Trace.Write(ex.Message);
    }
}

/// <summary>
/// 根据所存图片类型返回图片格式
/// </summary>
/// <param name="imageType">图片类型</param>
/// <param name="ret_type">返回的图片类型 (string)</param>
/// <returns>返回的图片类型 (ImageFormat)</returns>
private System.Drawing.Imaging.ImageFormat GetImageType(string imageType, out string ret_type)
{
    switch (imageType)
    {
        case "image/gif":
            ret_type = "gif";
            return System.Drawing.Imaging.ImageFormat.Gif;
            break;
        case "image/bmp":
            ret_type = "bmp";
            return System.Drawing.Imaging.ImageFormat.Bmp;
            break;
        case "image/x-png":
        case "image/png":
            ret_type = "png";
            return System.Drawing.Imaging.ImageFormat.Png;
            break;
        case "image/pjpeg":
        case "image/jpeg":
            ret_type = "jpg";
            return System.Drawing.Imaging.ImageFormat.Jpeg;
            break;
        default:
            ret_type = "jpg";
            return System.Drawing.Imaging.ImageFormat.Jpeg;
            break;
    }   
}

错误: 超时时间已到, 但是尚未从池中获取连接.

出现这种情况可能是因为所有池连接均在使用, 并且达到了最大池大小, 解决办法: 在数据库连接配置中加上 Max Pool Size = 512;.

合并两个表结构完全一致的 DataTable

/// <summary>
/// 将 sum 表和 para 表合并到 sum 表中, 要求: "sum 表" 和 "para 表" 的表结构必须完全一样!
/// </summary>
/// <param name="sum">待合并表之一, 同时也是结果表</param>
/// <param name="para">仅作为待合并表之一</param>
/// <returns></returns>
private static void AddTableToTable(DataTable sum, DataTable para)
{
    //遍历 para 表的每一行
    foreach (DataRow dataRow_para in para.Rows)
    {
        //新建 sum 表的一行
        DataRow dataRow_sum = sum.NewRow();

        //新建的这一行取 para 表的数据
        dataRow_sum.ItemArray = dataRow_para.ItemArray;

        //将新建的这一样加到 sum 表中
        sum.Rows.Add(dataRow_sum);
    }
}

{

        document.getElementById("btn_cx").click();

        return false; // 阻止 IE 浏览器的默认 Enter 行为
