{
    switch (imageType)
    {
        case "image/gif":
            ret_type = "gif";
            return System.Drawing.Imaging.ImageFormat.Gif;
            break;
        case "image/bmp":
            ret_type = "bmp";

            break;
        case "image/x-png":
        case "image/png":
            ret_type = "png";
            return System.Drawing.Imaging.ImageFormat.Png;
            break;
        case "image/pjpeg":
        case "image/jpeg":
            ret_type = "jpg";
            return System.Drawing.Imaging.ImageFormat.Jpeg;
            break;
        default:
            ret_type = "jpg";
            return System.Drawing.Imaging.ImageFormat.Jpeg;
            break;
    }   
}
